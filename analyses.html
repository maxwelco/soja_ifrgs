    cultivar == "59i58more" ~ "more",
    cultivar == "5892" ~ "normal",

    cultivar == "55i57" ~ "normal", 
    cultivar == "55i57less" ~ "less",
    cultivar == "55i57more" ~ "more",
    TRUE ~ NA_character_)) %>%

         injury = as_factor(injury),
